# Common production configurations:
# MySQL Example:
# DATABASE_URL="mysql://volunteer_user:your_password@localhost:3306/volunteer_db"

# PostgreSQL Example:
# DATABASE_URL="postgresql://volunteer_user:your_password@localhost:5432/volunteer_db"

# Replace with your actual production database credentials:
# - username: Your database username
# - password: Your database password  
# - host: Your database host (localhost or remote server)
# - port: Database port (3306 for MySQL, 5432 for PostgreSQL)
# - database_name: Your production database name
